250 GLuint ShaderProgram::loadShaderFromFile( std::string path, GLenum shaderType )
251 {
252  //Open file
253  GLuint shaderID = 0;
254  std::string shaderString;
255  std::ifstream sourceFile( path.c_str() );
256 
257  //Source file loaded
258  if( sourceFile )
259  {
260  //Get shader source
261  shaderString.assign( ( std::istreambuf_iterator< char >( sourceFile ) ), std::istreambuf_iterator< char >() );
262 
263  //Create shader ID
264  shaderID = glCreateShader( shaderType );
265 
266  //Set shader source
267  const GLchar* shaderSource = shaderString.c_str();
268  glShaderSource( shaderID, 1, (const GLchar**)&shaderSource, NULL );
269 
270  //Compile shader source
271  glCompileShader( shaderID );
272 
273  //Check shader for errors
274  GLint shaderCompiled = GL_FALSE;
275  glGetShaderiv( shaderID, GL_COMPILE_STATUS, &shaderCompiled );
276  if( shaderCompiled != GL_TRUE )
277  {
278  printf( "Unable to compile shader %d!\n\nSource:\n%s\n", shaderID, shaderSource );
279  printShaderLog( shaderID );
280  glDeleteShader( shaderID );
281  shaderID = 0;
282  }
283  }
284  else
285  {
286  printf( "Unable to open file %s\n", path.c_str() );
287  }
288 
289  return shaderID;
290 }
