217 bool BasicTexturedPolygonShader::loadProgram()
218 {
219  //Generate program
220  mProgramID = glCreateProgram();
221 
222  //Load vertex shader
223  GLuint vertexShader = loadShaderFromFile( "shaders/basicPolygon.glvs", GL_VERTEX_SHADER );
224 
225  //Check for errors
226  if( vertexShader == 0 )
227  {
228  glDeleteProgram( mProgramID );
229  mProgramID = 0;
230  return false;
231  }
232 
233  //Attach vertex shader to program
234  glAttachShader( mProgramID, vertexShader );
235 
236 
237  //Create fragment shader
238  GLuint fragmentShader = loadShaderFromFile( "shaders/basicPolygon.glfs", GL_FRAGMENT_SHADER );
239 
240  //Check for errors
241  if( fragmentShader == 0 )
242  {
243  glDeleteShader( vertexShader );
244  glDeleteProgram( mProgramID );
245  mProgramID = 0;
246  return false;
247  }
248 
249  //Attach fragment shader to program
250  glAttachShader( mProgramID, fragmentShader );
251 
252  //Link program
253  glLinkProgram( mProgramID );
254 
255  //Check for errors
256  GLint programSuccess = GL_TRUE;
257  glGetProgramiv( mProgramID, GL_LINK_STATUS, &programSuccess );
258  if( programSuccess != GL_TRUE )
259  {
260  cout << "Error linking program " << mProgramID << "!\n";
261  printProgramLog( mProgramID );
262  glDeleteShader( vertexShader );
263  glDeleteShader( fragmentShader );
264  glDeleteProgram( mProgramID );
265  mProgramID = 0;
266  return false;
267  }
268 
269  loadUniforms();
270 
271  //Clean up excess shader references
272  glDeleteShader( vertexShader );
273  glDeleteShader( fragmentShader );
274 
275  return true;
276 }
