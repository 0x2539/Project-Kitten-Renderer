183 void Stars::update(){
184 
185  for(unsigned int i = 0; i < stars.size(); ++i)
186  {
187  float distance = dist(_CenterX, _CenterY, stars[i] -> getLocationX(), stars[i] -> getLocationY());
188  float angle = starsInitialAngle[i] + starsRotationSpeed[i];
189 
190  if(angle >= 360) angle -= 360;
191  if(angle < 0) angle += 360;
192

194  float y = _CenterY + distance * sin(angle * M_PI / 180);
195 
196  starsInitialAngle[i] = angle;

198  stars[i] -> setLocation(x, y);
199  }
