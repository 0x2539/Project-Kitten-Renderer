311 void ShapeRectangle::setGraphicBuffers()
312 {
313  GLfloat texLeft = 0;
314  GLfloat texTop = 0;
315  GLfloat texRight = 1;
316  GLfloat texBottom = 1;
317 
318  vertices[ 0 ].setVertexCoord( GLVertexCoord( 0, 0 ) );
319  vertices[ 1 ].setVertexCoord( GLVertexCoord( getWidth(), 0 ) );
320  vertices[ 2 ].setVertexCoord( GLVertexCoord( getWidth(), getHeight() ) );
321  vertices[ 3 ].setVertexCoord( GLVertexCoord( 0, getHeight() ) );
322 
323  vertices[ 0 ].setTextureCoord( GLTexCoord( texLeft, texTop ) );
324  vertices[ 1 ].setTextureCoord( GLTexCoord( texRight, texTop ) );
325  vertices[ 2 ].setTextureCoord( GLTexCoord( texRight, texBottom ) );
326  vertices[ 3 ].setTextureCoord( GLTexCoord( texLeft, texBottom ) );
327 
328  //Set rendering indices
329  indices[ 0 ] = 0;
330  indices[ 1 ] = 1;
331  indices[ 2 ] = 2;
332  indices[ 3 ] = 3;
333 
334  //glBindTexture( GL_TEXTURE_2D, GLUtils::texture );
335  //Create VBO
336  glGenBuffers( 1, &gVBO );
337  glBindBuffer( GL_ARRAY_BUFFER, gVBO );
338  glBufferData( GL_ARRAY_BUFFER, 4 * sizeof(GLTexturedVertex), vertices, GL_DYNAMIC_DRAW );
339 
340  //Create IBO
341  glGenBuffers( 1, &gIBO );
342  glBindBuffer( GL_ELEMENT_ARRAY_BUFFER, gIBO );
343  glBufferData( GL_ELEMENT_ARRAY_BUFFER, 4 * sizeof(GLuint), indices, GL_DYNAMIC_DRAW );
344 
345  //Unbind buffers
346  glBindBuffer( GL_ARRAY_BUFFER, NULL );
347  glBindBuffer( GL_ELEMENT_ARRAY_BUFFER, NULL );
348 }

365 void ShapeRectangle::draw(){
366 
367  // if it's not currently on the screen (actually camera doesn't see it), don't draw it
368  if(! Camera::isVisible(this -> getLocationX(), this -> getLocationY(), this -> getWidth(), this -> getHeight()))
369  return ;
370 
371  setColor(0.5f, 1.f, 0.5f, 1.0f);
372 
373  updateVertices();
374 
375  basicTexturedPolygonShader->setModelView( glm::translate<GLfloat>(
376  glm::vec3( this -> getLocationX() - Camera::getX(),
377  this -> getLocationY() - Camera::getY(), 0.f ) ) );
378  basicTexturedPolygonShader->updateModelView();
379 
380  //Set texture ID
381  glBindTexture( GL_TEXTURE_2D, texture );
382 
383  basicTexturedPolygonShader->enableVertexPointer();
384  basicTexturedPolygonShader->enableTexCoordPointer();
385  //Enable vertex arrays
386  //glEnableClientState( GL_VERTEX_ARRAY );
387 
388  //Set vertex data
389  glBindBuffer( GL_ARRAY_BUFFER, gVBO );
390  //Update vertex buffer data
391  glBufferSubData( GL_ARRAY_BUFFER, 0, 4 * sizeof(GLTexturedVertex), vertices );
392 
393  //Set texture coordinate data
394  basicTexturedPolygonShader->setTexCoordPointer( sizeof(GLTexturedVertex), (GLvoid*)offsetof( GLTexturedVertex, _textureCoord ) );
395 
396  //Set vertex data
397  basicTexturedPolygonShader->setVertexPointer( sizeof(GLTexturedVertex), (GLvoid*)offsetof( GLTexturedVertex, _vertexCoord ) );
398 
399 
400  //Draw quad using vertex data and index data
401  glBindBuffer( GL_ELEMENT_ARRAY_BUFFER, gIBO );
402  glDrawElements( GL_TRIANGLE_FAN, 4, GL_UNSIGNED_INT, NULL );
403 
404  //Disable vertex arrays
405  //glDisableClientState( GL_VERTEX_ARRAY );
406  //Disable vertex and texture coordinate arrays
407  basicTexturedPolygonShader->disableVertexPointer();
408  basicTexturedPolygonShader->disableTexCoordPointer();
409 }
